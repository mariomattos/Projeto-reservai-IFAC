        }

    ID RF10 CADASTRAR ESPAÇO {
        Descrição {
            Permite a inclusão de um novo ambiente físico utilizável na base de dados da aplicação.
        }

        Restrições {
            Acesso restrito ao Administrador. O espaço não deve possuir o mesmo nome ou identificador de um já existente.
        }

        Ações {
            Preencher os dados obrigatórios do ambiente (Nome, capacidade, localização) e salvar.
        }

        Resultado {
            O sistema registrará o novo ambiente físico no banco de dados e o disponibilizará no catálogo de agendamentos, caso contrário, exibirá uma mensagem descrevendo o erro de duplicidade ou inconsistência nos dados de entrada.
        }
    } //Fim RF10 CADASTRAR ESPAÇO

    ID RF11 LISTAR CADASTROS DE ESPAÇOS {
        Descrição {
            Exibe um catálogo ou lista com todos os espaços e ambientes físicos registrados no sistema.
        }

        Restrições {
            Nenhuma. Visível para todos os tipos de usuários da plataforma.
        }

        Ações {
            Acessar a seção de ambientes ou espaços no menu de navegação.
        }

        Resultado {
            O sistema renderizará a lista de espaços e suas principais características físicas em tela, caso contrário, exibirá uma mensagem de erro descrevendo a impossibilidade de carregar os dados no momento.
        }
    } //Fim RF11 LISTAR CADASTROS DE ESPAÇOS

    ID RF12 CONSULTAR CADASTRO DE ESPAÇO {
        Descrição {
            Apresenta as informações completas e detalhadas sobre a infraestrutura e regras de um espaço específico.
        }

        Restrições {
            O espaço selecionado deve estar cadastrado no banco de dados.
        }

        Ações {
            Selecionar o espaço desejado a partir da listagem geral ou através de busca direta.
        }

        Resultado {
            O sistema exibirá a página com especificações completas, fotos, capacidade e recursos do ambiente focado, caso contrário, exibirá uma mensagem alertando que o espaço está indisponível ou inexistente.
        }
    } //Fim RF12 CONSULTAR CADASTRO DE ESPAÇO

    ID RF13 ALTERAR CADASTRO DE ESPAÇO {
        Descrição {
            Permite atualizar as informações, capacidades e detalhes físicos de um ambiente existente.
        }

        Restrições {
            Acesso exclusivo ao perfil de Administrador.
        }

        Ações {
            Modificar os dados do ambiente na tela de edição correspondente e confirmar o salvamento.
        }

        Resultado {
            O sistema atualizará as configurações e a infraestrutura do espaço em sua base, caso contrário, exibirá uma mensagem descrevendo o erro cometido na edição dos campos ou valores inadequados.
        }
    } //Fim RF13 ALTERAR CADASTRO DE ESPAÇO

    ID RF14 EXCLUIR CADASTRO DE ESPAÇO {
        Descrição {
            Remove em definitivo ou desativa as operações de agendamento de um ambiente do sistema.
        }

        Restrições {
            Acesso restrito ao Administrador. O espaço não pode possuir reservas ativas ou pendentes no calendário futuro.
        }

        Ações {
            Acionar a opção de exclusão dentro do painel do ambiente e confirmar o aviso de segurança.
        }

        Resultado {
            O sistema removerá o ambiente físico do catálogo público e indisponibilizará futuras consultas, caso contrário, exibirá uma mensagem alertando que o local possui agendamentos ativos vinculados.
        }
    } //Fim RF14 EXCLUIR CADASTRO DE ESPAÇO

    ID RF15 VISUALIZAR FOTOS DO ESPAÇO {
        Descrição {
            Exibe uma galeria de imagens vinculadas ao espaço para conferência de sua estrutura física.
        }

        Restrições {
            O espaço consultado deve conter mídias/imagens previamente anexadas pelo administrador.
        }

        Ações {
            Clicar na seção de fotos ou sobre o carrossel de imagens contido nos detalhes do ambiente.
        }

        Resultado {
            O sistema expandirá o carrossel de fotos exibindo as mídias do ambiente em tela cheia, caso contrário, exibirá uma imagem padrão indicando falha no carregamento dos arquivos anexados.
        }
    } //Fim RF15 VISUALIZAR FOTOS DO ESPAÇO

    ID RF16 FAVORITAR ESPAÇO {
        Descrição {
            Permite marcar ambientes específicos para rápido acesso posterior através de uma lista personalizada.
        }

        Restrições {
            O usuário deve estar autenticado no sistema.
        }

        Ações {
            Clicar no ícone indicador de favoritos (ex: estrela ou coração) presente no card ou página do espaço.
        }

        Resultado {
            O sistema associará e indexará o espaço na lista de preferências rápidas da conta do usuário, caso contrário, exibirá uma mensagem informando que houve um erro na sincronização da ação.
        }
    } //Fim RF16 FAVORITAR ESPAÇO


    // ==========================================
    // BLOCO 3: RESERVAS E AGENDAMENTOS
    // ==========================================

    ID RF17 CADASTRAR RESERVA DE ESPAÇO {
        Descrição {
            Registra uma nova intenção de uso para um espaço físico em data e horários determinados.
        }

        Restrições {
            O horário e o espaço escolhidos devem estar totalmente livres e livres de bloqueios na agenda.
        }

        Ações {
            Selecionar o ambiente, definir data/hora de início e término e confirmar a solicitação.
        }

        Resultado {
            O sistema criará o agendamento no calendário e o submeterá para a fila de validação, caso contrário, exibirá uma mensagem informando que há conflito de horários para o intervalo selecionado.
        }
    } //Fim RF17 CADASTRAR RESERVA DE ESPAÇO

    ID RF18 ALTERAR RESERVA DE ESPAÇO {
        Descrição {
            Modifica o período ou o local de uma reserva previamente registrada pelo próprio solicitante ou pelo administrador.
        }

        Restrições {
            O novo período pretendido deve estar disponível na agenda do ambiente físico correspondente.
        }

        Ações {
            Acessar a gerência de reservas, editar os parâmetros de tempo/espaço e confirmar a troca.
        }

        Resultado {
            O sistema atualizará os horários ou o ambiente físico daquela reserva no calendário central, caso contrário, exibirá uma mensagem informando que a modificação colide com restrições institucionais ou reservas já existentes.
        }
    } //Fim RF18 ALTERAR RESERVA DE ESPAÇO

    ID RF19 CANCELAR RESERVA DE ESPAÇO {
        Descrição {
            Cancela e invalida uma reserva ativa, liberando o horário correspondente no calendário para novos agendamentos.
        }

        Restrições {
            O cancelamento deve obedecer ao limite de antecedência mínima configurado nas regras do sistema.
        }

        Ações {
            Selecionar a reserva desejada no menu pessoal e acionar o comando de cancelamento.
        }

        Resultado {
            O sistema invalidará a reserva e mudará seu status para cancelado liberando o horário na agenda, caso contrário, exibirá uma mensagem informando que o prazo de tolerância para cancelamentos expirou.
        }
    } //Fim RF19 CANCELAR RESERVA DE ESPAÇO

    ID RF20 LISTAR RESERVAS DE ESPAÇOS {
        Descrição {
            Lista os registros de agendamentos efetuados no sistema, exibindo solicitantes, locais e períodos temporais.
        }

        Restrições {
            Usuários comuns visualizam apenas as suas reservas; Administradores visualizam a totalidade das reservas.
        }

        Ações {
            Navegar até a tela de controle ou histórico de agendamentos.
        }

        Resultado {
            O sistema apresentará uma tabela com todos os agendamentos e seus respectivos status de processamento, caso contrário, exibirá uma mensagem informando falha de autorização ou indisponibilidade de dados.
        }
    } //Fim RF20 LISTAR RESERVAS DE ESPAÇOS

    ID RF21 CONSULTAR RESERVA DE ESPAÇO {
        Descrição {
            Exibe de forma detalhada as informações de uma reserva específica, incluindo histórico de aprovação e justificativas.
        }

        Restrições {
            O usuário deve possuir a reserva sob sua titularidade ou deter permissão administrativa.
        }

        Ações {
            Clicar sobre uma reserva a partir do histórico ou por meio de notificação interna.
        }

        Resultado {
            O sistema renderizará a tela com os dados completos de tempo, justificativa e o histórico da reserva, caso contrário, exibirá uma mensagem informando que o acesso a este agendamento é restrito ou inválido.
        }
    } //Fim RF21 CONSULTAR RESERVA DE ESPAÇO

    ID RF22 CONSULTAR AGENDA DE RESERVAS DE UM ESPAÇO {
        Descrição {
            Apresenta uma visão unificada e cronológica (calendário semanal/mensal) de ocupação de um ambiente específico.
        }

        Restrições {
            Nenhuma. A agenda pública deve estar disponível para fins de transparência de horários.
        }

        Ações {
            Acessar a aba de agenda/calendário contida nas especificações do espaço desejado.
        }

        Resultado {
            O sistema exibirá a grade de horários visual com a marcação dos blocos livres e ocupados daquele ambiente, caso contrário, exibirá uma mensagem descrevendo a falha de comunicação com o microsserviço de calendário.
        }
    } //Fim RF22 CONSULTAR AGENDA DE RESERVAS DE UM ESPAÇO

    ID RF23 ENTRAR NA LISTA DE ESPERA PRA RESERVA {
        Descrição {
            Adiciona o usuário a uma fila ordenada de prioridade caso o horário desejado em determinado espaço já esteja ocupado.
        }

        Restrições {
            O período selecionado precisa estar obrigatoriamente ocupado por outra reserva ativa de terceiro.
        }

        Ações {
            Solicitar a inclusão na fila de espera na tentativa de reservar um bloco temporal indisponível.
        }

        Resultado {
            O sistema registrará o usuário na posição sequencial correta da fila de espera daquele horário, caso contrário, exibirá uma mensagem alertando que a inclusão falhou por já haver uma solicitação idêntica e ativa.
        }
    } //Fim RF23 ENTRAR NA LISTA DE ESPERA PRA RESERVA


    // ==========================================
    // BLOCO 4: CONTROLE ADMINISTRATIVO DE RESERVAS
    // ==========================================

    ID RF24 BLOQUEAR RESERVA DO ESPAÇO POR PERÍODO {
        Descrição {
            Interrompe de forma temporária a agenda de agendamentos de um local físico por razões técnicas ou institucionais.
        }

        Restrições {
            Acesso exclusivo ao perfil de Administrador.
        }

        Ações {
            Definir o local, o intervalo de tempo de início/fim do bloqueio e salvar a restrição.
        }

        Resultado {
            O sistema invalidará novas reservas normais para o período e marcará o intervalo como bloqueado na agenda, caso contrário, exibirá uma mensagem alertando que a data de início informada é retroativa ou inválida.
        }
    } //Fim RF24 BLOQUEAR RESERVA DO ESPAÇO POR PERÍODO

    ID RF25 DESBLOQUEAR RESERVAS DO ESPAÇO {
        Descrição {
            Revoga um bloqueio temporário aplicado previamente a um espaço, reestabelecendo a normalidade de sua agenda.
        }

        Restrições {
            Acesso exclusivo ao Administrador. O período deve possuir um bloqueio ativo de antemão.
        }

        Ações {
            Selecionar o bloqueio em vigor na listagem de restrições do ambiente e acionar a liberação.
        }

        Resultado {
            O sistema removerá a restrição de calendário e reabrirá o ambiente para recebimento de novas requisições, caso contrário, exibirá uma mensagem indicando que o bloqueio não foi localizado ou já expirou.
        }
    } //Fim RF25 DESBLOQUEAR RESERVAS DO ESPAÇO

    ID RF26 APROVAR RESERVA DO ESPAÇO {
        Descrição {
            Valida e concede a autorização oficial de uso para um agendamento que se encontra no estado pendente.
        }

        Restrições {
            Acesso restrito ao perfil Administrador. A reserva destino deve estar aguardando triagem.
        }

        Ações {
            Acessar a fila de triagem de agendamentos e clicar na opção de aprovação do registro.
        }

        Resultado {
            O sistema alterará o status da reserva para aprovado e notificará o solicitante automaticamente, caso contrário, exibirá uma mensagem alertando que o agendamento possui inconsistências cronológicas ou passou do horário útil.
        }
    } //Fim RF26 APROVAR RESERVA DO ESPAÇO

    ID RF27 REPROVAR RESERVA DO ESPAÇO {
        Descrição {
            Nega formalmente uma requisição pendente de reserva efetuada por um usuário da plataforma.
        }

        Restrições {
            Acesso restrito ao Administrador. Exige o preenchimento de uma justificativa textual para a negativa.
        }

        Ações {
            Selecionar a opção de rejeição na fila de triagem e inserir a razão da recusa.
        }

        Resultado {
            O sistema salvará a recusa modificando o status da reserva para reprovado e enviando o feedback ao solicitante, caso contrário, exibirá uma mensagem obrigando o preenchimento do campo de justificativa.
        }
    } //Fim RF27 REPROVAR RESERVA DO ESPAÇO

    ID RF28 DEFINIR REGRAS DE RESERVA POR ESPAÇO {
        Descrição {
            Estabelece parâmetros de uso específicos para cada local (antecedência, perfis autorizados e limite de horas).
        }

        Restrições {
            Permissão exclusiva de configuração dada a usuários com perfil de Administrador.
        }

        Ações {
            Acessar o menu de políticas do ambiente, configure as restrições desejadas e salve o conjunto de regras.
        }

        Resultado {
            O sistema atualizará as diretrizes de governança e validação de agendamentos aplicadas àquele local, caso contrário, exibirá uma mensagem indicando incoerência ou erro de configuração matemática nos parâmetros de tempo.
        }
    } //Fim RF28 DEFINIR REGRAS DE RESERVA POR ESPAÇO

    ID RF29 REALIZAR CHECK-IN DE RESERVA {
        Descrição {
            Confirma eletronicamente a presença física e o início de uso efetivo do espaço alocado.
        }

        Restrições {
            Deve ser executado dentro do raio geográfico ou na janela de tempo permitida para o início da reserva.
        }

        Ações {
            Acionar o botão de check-in na reserva activa ou efetuar a leitura do código de presença no local (ex: QR Code).
        }

        Resultado {
            O sistema atualizará a reserva para o status em utilização e gravará a hora de entrada do usuário, caso contrário, exibirá uma mensagem informando que a janela de tempo ou a validação de presença falhou.
        }
    } //Fim RF29 REALIZAR CHECK-IN DE RESERVA

    ID RF30 REALIZAR CHECK-OUT DE RESERVA {
        Descrição {
            Registra formalmente o encerramento da utilização do espaço físico, liberando-o de forma antecipada ou pontual.
        }

        Restrições {
            A reserva correspondente deve estar com o status de em utilização ativo no sistema.
        }

        Ações {
            Confirmar a saída do ambiente através da interface do aplicativo ou encerramento manual.
        }

        Resultado {
            O sistema finalizará a contagem de tempo da alocação mudando o status da reserva para concluída, caso contrário, exibirá uma mensagem indicando erro ao computar a saída ou duplicidade de requisição.
        }
    } //Fim RF30 REALIZAR CHECK-OUT DE RESERVA

        }

        Restrições {
            O sistema deve estar operando sob o esquema de cores padrão ou claro.
        }

        Ações {

        }

        Resultado {

    } //Fim RF33 ALTERAR PARA O MODO ESCURO

        Descrição {

        }

        Restrições {
            O sistema deve estar operando sob o esquema de cores do modo escuro.
        }

        Ações {
            Selecionar a opção correspondente ao Modo Claro nas configurações de personalização visual.
        }

        Resultado {
            O sistema modificará a interface aplicando os estilos e esquemas de cores de alta iluminação padrão, caso contrário, exibirá uma mensagem sugerindo a atualização manual da página por travamento de renderização.
        }
    } //Fim RF34 ALTERAR PARA O MODO CLARO

        Descrição {
            Aumenta ou reduz de forma proporcional a escala de tamanho dos textos exibidos nas interfaces do software.
        }

        Restrições {
            O redimensionamento deve respeitar as proporções mínimas e máximas de legibilidade de layout definidas.
        }

        Ações {
            Utilizar os controles de tamanho de fonte nas opções de acessibilidade para ajustar o nível do texto.
        }

        Resultado {
            O sistema ajustará de forma harmônica a tipografia de todas as telas conforme o nível escolhido, caso contrário, manterá o tamanho atual emitindo uma mensagem de que o limite crítico de layout foi atingido.
        }
    } //Fim RF35 DEFINIR ESCALONAMENTO DE FONTE

    ID RF36 AVALIAR ESPAÇO {
        Descrição {
            Permite registrar uma nota numérica e comentários sobre a experiência de uso de um ambiente físico.
        }

        Restrições {
            O usuário deve ter completado e realizado com sucesso o check-out de uma reserva prévia naquele ambiente.
        }

        Ações {
            Atribuir uma nota em escala estrelar, redigir uma crítica opcional e enviar a avaliação no formulário.
        }

        Resultado {
            O sistema computará a pontuação informada e recalculará a média geral de reputação do local físico, caso contrário, exibirá uma mensagem de erro alertando que o envio foi bloqueado por ausência de check-out válido no histórico.
        }
    } //Fim RF36 AVALIAR ESPAÇO

    ID RF37 EDITAR AVALIAÇÃO DO ESPAÇO {
        Descrição {
            Permite ao autor de uma crítica cadastrada modificar o teor de seu texto ou nota de pontuação.
        }

        Restrições {
            A alteração só pode ser efetuada na avaliação de propriedade do próprio usuário autenticado.
        }

        Ações {
            Localizar a avaliação enviada no histórico de opiniões, alterar os campos e salvar.
        }

        Resultado {
            O sistema substituirá a nota e o texto descritivo anteriores pelos novos dados preenchidos na base, caso contrário, exibirá uma mensagem informando que a crítica foi bloqueada ou moderada previamente, impossibilitando a edição.
        }
    } //Fim RF37 EDITAR AVALIAÇÃO DO ESPAÇO

    ID RF38 LISTAR AVALIAÇÕES DE ESPAÇOS {
        Descrição {
            Gera uma exibição consolidada de todas as pontuações e relatos deixados pelos frequentadores dos ambientes.
        }

        Restrições {
            Devem existir avaliações registradas na base de dados para que a listagem não retorne vazia.
        }

        Ações {
            Acessar a seção pública de avaliações gerais ou o painel gerencial correspondente.
        }

        Resultado {
            O sistema renderizará a linha do tempo cronológica com o feedback textual e as estrelas recebidas pelos ambientes, caso contrário, exibirá uma mensagem descrevendo falha de conexão na busca de relatos.
        }
    } //Fim RF38 LISTAR AVALIAÇÕES DE ESPAÇOS

    ID RF39 CONSULTAR AVALIAÇÃO DO ESPAÇO {
        Descrição {
            Filtra e exibe detalhadamente as notas e opiniões dadas exclusivamente sobre um único ambiente selecionado.
        }

        Restrições {
            O espaço alvo do filtro deve estar devidamente identificado na consulta.
        }

        Ações {
            Abrir a aba de avaliações inserida na página de exibição de dados de um determinado espaço físico.
        }

        Resultado {
            O sistema isolará e exibirá o conjunto de reviews e notas atribuídos especificamente ao local pesquisado, caso contrário, apresentará uma caixa de busca limpa indicando que o filtro de busca aplicado é inválido.
        }
    } //Fim RF39 CONSULTAR AVALIAÇÃO DO ESPAÇO

    ID RF40 ABRIR CHAT INTERNO ENTRE USUÁRIO E ADMIN {
        Descrição {
            Inicia um canal de mensagens textuais diretas e em tempo real para comunicação de suporte e esclarecimentos.
        }

        Restrições {
            O usuário necessita estar autenticado na plataforma para estabelecer a comunicação de suporte.
        }

        Ações {
            Acessar a central de ajuda ou mensagens e redigir uma mensagem de abertura de atendimento.
        }

        Resultado {
            O sistema abrirá a sala de bate-papo indexando a conversa à mesa de suporte da administração de TI, caso contrário, exibirá uma mensagem alertando sobre falha na conexão do canal de chat em tempo real.
        }
    } //Fim RF40 ABRIR CHAT INTERNO ENTRE USUÁRIO E ADMIN

    ID RF41 REPORTAR PROBLEMA NO ESPAÇO {
        Descrição {
            Permite notificar falhas estruturais, de limpeza ou danos em equipamentos de um ambiente para manutenção técnica.
        }

        Restrições {
            O usuário deve relatar falhas relacionadas a um ambiente físico existente no catálogo do sistema.
        }

        Ações {
            Preencher o formulário de incidentes do espaço detalhando o ocorrido e opcionalmente anexar fotos.
        }

        Resultado {
            O sistema criará uma ordem de serviço de reparo encaminhando o chamado para a gerência operacional, caso contrário, exibirá uma mensagem informando insuficiência de caracteres na descrição ou preenchimento incorreto.
        }
    } //Fim RF41 REPORTAR PROBLEMA NO ESPAÇO

    ID RF42 LIMITAR RESERVAS SIMULTÂNEAS POR USUÁRIO/ESPAÇO {
        Descrição {
            Garante a aplicação de regras computacionais automatizadas que travam agendamentos abusivos ou em paralelo por um mesmo indivíduo.
        }

        Restrições {
            A regra deve ser validada sistemicamente pelo motor de regras em toda tentativa de criação de reserva (`RF17`).
        }

        Ações {
            O sistema faz a contagem de agendamentos em aberto do usuário no momento em que ele submete um novo pedido.
        }

        Resultado {
            O sistema aceitará o processamento da reserva por estar dentro do quantitativo de cota permitida, caso contrário, bloqueará a transação exibindo uma mensagem notificando que o teto limite máximo de reservas do usuário foi alcançado.
        }
    } //Fim RF42 LIMITAR RESERVAS SIMULTÂNEAS POR USUÁRIO/ESPAÇO
